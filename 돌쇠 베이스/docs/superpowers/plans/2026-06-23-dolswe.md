# dolswe/shared_state.py
import threading

class SharedContext:
    def __init__(self):
        self._lock = threading.Lock()
        self._data = {"people_count": 0, "gesture": None, "color": None, "scene": None}

    def update(self, **kwargs):
        with self._lock:
            self._data.update(kwargs)

    def snapshot(self) -> dict:
        with self._lock:
            return dict(self._data)
```

# dolswe/color.py
import cv2
import numpy as np
from dolswe import config

def dominant_color_name(bgr_region):
    if bgr_region is None or bgr_region.size == 0:
        return None
    hsv = cv2.cvtColor(bgr_region, cv2.COLOR_BGR2HSV)
    h = int(np.median(hsv[:, :, 0]))
    s = int(np.median(hsv[:, :, 1]))
    v = int(np.median(hsv[:, :, 2]))
    if v < config.BLACK_V:
        return "검은"
    if s < 50:
        return "흰" if v > config.WHITE_V else config.GRAY_NAME
    for h_low, h_high, s_min, v_min, name in config.COLOR_TABLE:
        if h_low <= h <= h_high and s >= s_min and v >= v_min:
            return name
    return config.GRAY_NAME
```

# dolswe/gesture.py
def is_waving(x_positions, min_reversals=3, min_amplitude=0.05):
    xs = list(x_positions)
    if len(xs) < 4:
        return False
    if max(xs) - min(xs) < min_amplitude:
        return False
    reversals = 0
    direction = 0
    for prev, cur in zip(xs, xs[1:]):
        d = cur - prev
        if abs(d) < min_amplitude / 4:
            continue
        sign = 1 if d > 0 else -1
        if direction and sign != direction:
            reversals += 1
        direction = sign
    return reversals >= min_reversals
```

# dolswe/sentences.py
_ENDERS = ".!?…\n"

class SentenceSplitter:
    def __init__(self, max_len=60):
        self._buf = ""
        self._max_len = max_len

    def feed(self, token: str) -> list[str]:
        self._buf += token
        out = []
        while True:
            idx = next((i for i, c in enumerate(self._buf) if c in _ENDERS), -1)
            if idx >= 0:
                sentence = self._buf[: idx + 1].strip()
                self._buf = self._buf[idx + 1 :]
                if sentence:
                    out.append(sentence)
                continue
            if len(self._buf) >= self._max_len:
                out.append(self._buf.strip())
                self._buf = ""
                continue
            break
        return out

    def flush(self):
        rem = self._buf.strip()
        self._buf = ""
        return rem or None
```

# dolswe/audio_utils.py
import numpy as np

def amplitude_envelope(samples, n_bins=20):
    raw = np.asarray(samples)
    arr = raw.astype(np.float64)
    if arr.size == 0:
        return [0.0] * n_bins
    if np.issubdtype(raw.dtype, np.integer):
        arr = arr / 32768.0
    bins = np.array_split(arr, n_bins)
    out = []
    for b in bins:
        rms = float(np.sqrt(np.mean(b ** 2))) if b.size else 0.0
        out.append(min(1.0, rms))
    return out
```

# dolswe/brain.py
from dolswe import config
from dolswe.persona import SYSTEM_PROMPT, build_context_line
from dolswe.sentences import SentenceSplitter

def build_messages(history, user_text, snapshot):
    msgs = [{"role": "system", "content": SYSTEM_PROMPT}]
    ctx = build_context_line(snapshot)
    if ctx:
        msgs.append({"role": "system", "content": ctx})
    msgs.extend(history)
    msgs.append({"role": "user", "content": user_text})
    return msgs

class Brain:
    def __init__(self, client=None, max_history=12):
        if client is None:
            import ollama
            client = ollama
        self._client = client
        self.history = []
        self._max_history = max_history

    def respond(self, user_text, snapshot):
        messages = build_messages(self.history, user_text, snapshot)
        splitter = SentenceSplitter()
        full = ""
        for chunk in self._client.chat(model=config.LLM_MODEL, messages=messages, stream=True):
            piece = chunk["message"]["content"]
            full += piece
            for sentence in splitter.feed(piece):
                yield sentence
        rem = splitter.flush()
        if rem:
            yield rem
        self.history.append({"role": "user", "content": user_text})
        self.history.append({"role": "assistant", "content": full})
        self.history = self.history[-self._max_history :]
```

```python
# dolswe/stt.py
import threading
import queue
import numpy as np
import sounddevice as sd
from faster_whisper import WhisperModel
from dolswe import config

SAMPLE_RATE = 16000
BLOCK = 1600          # 0.1s
SILENCE_RMS = 0.01    # 발화 끝 판정 임계
SILENCE_BLOCKS = 8    # 0.8s 무음이면 발화 종료

class SttWorker:
    def __init__(self, on_partial, on_final, mic_index=None):
        self._on_partial = on_partial
        self._on_final = on_final
        self._mic_index = mic_index
        self._model = WhisperModel(config.STT_MODEL, device="cpu", compute_type="int8")
        self._audio_q = queue.Queue()
        self._running = False
        self._thread = None

    def start(self):
        self._running = True
        self._thread = threading.Thread(target=self._loop, daemon=True)
        self._thread.start()

    def stop(self):
        self._running = False

    def _callback(self, indata, frames, time_info, status):
        self._audio_q.put(indata.copy())

    def _loop(self):
        buf = []
        silence = 0
        with sd.InputStream(samplerate=SAMPLE_RATE, channels=1, dtype="float32",
                            blocksize=BLOCK, device=self._mic_index, callback=self._callback):
            while self._running:
                try:
                    block = self._audio_q.get(timeout=0.5)
                except queue.Empty:
                    continue
                rms = float(np.sqrt(np.mean(block ** 2)))
                if rms >= SILENCE_RMS:
                    buf.append(block)
                    silence = 0
                elif buf:
                    silence += 1
                    buf.append(block)
                    if silence >= SILENCE_BLOCKS:
                        self._transcribe(np.concatenate(buf).flatten())
                        buf, silence = [], 0

    def _transcribe(self, audio):
        segments, _ = self._model.transcribe(audio, language="ko", beam_size=1)
        text = "".join(s.text for s in segments).strip()
        if text:
            self._on_partial(text)
            self._on_final(text)
```

# dolswe/vision_fast.py
import threading, time, collections
import cv2
import mediapipe as mp
from dolswe import config
from dolswe.color import dominant_color_name
from dolswe.gesture import is_waving

class VisionFast:
    def __init__(self, ctx, get_frame, on_new_person):
        self._ctx = ctx
        self._get_frame = get_frame
        self._on_new_person = on_new_person
        self._running = False
        self._thread = None
        self._face = mp.solutions.face_detection.FaceDetection(min_detection_confidence=0.5)
        self._hands = mp.solutions.hands.Hands(max_num_hands=1, min_detection_confidence=0.5)
        self._hand_x = collections.deque(maxlen=12)
        self._prev_count = 0
        self._last_trigger = 0.0

    def start(self):
        self._running = True
        self._thread = threading.Thread(target=self._loop, daemon=True)
        self._thread.start()

    def stop(self):
        self._running = False

    def _loop(self):
        interval = 1.0 / config.VISION_FAST_FPS
        while self._running:
            t0 = time.time()
            frame = self._get_frame()
            if frame is not None:
                rgb = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)
                faces = self._face.process(rgb)
                count = len(faces.detections) if faces.detections else 0
                color = None
                if count:
                    box = faces.detections[0].location_data.relative_bounding_box
                    color = self._body_color(frame, box)
                hands = self._hands.process(rgb)
                gesture = None
                if hands.multi_hand_landmarks:
                    self._hand_x.append(hands.multi_hand_landmarks[0].landmark[0].x)
                    if is_waving(self._hand_x):
                        gesture = "wave"
                else:
                    self._hand_x.clear()
                self._ctx.update(people_count=count, color=color, gesture=gesture)
                if count > self._prev_count and (time.time() - self._last_trigger) > config.NEW_PERSON_COOLDOWN:
                    self._last_trigger = time.time()
                    self._on_new_person()
                self._prev_count = count
            dt = time.time() - t0
            if dt < interval:
                time.sleep(interval - dt)

    def _body_color(self, frame, box):
        h, w = frame.shape[:2]
        # 얼굴 박스 아래(상체) 영역 색 추출
        x = max(0, int(box.xmin * w))
        y = min(h - 1, int((box.ymin + box.height) * h))
        x2 = min(w, int((box.xmin + box.width) * w))
        y2 = min(h, y + int(box.height * h * 1.5))
        return dominant_color_name(frame[y:y2, x:x2])

**Interfaces:**

import os, threading, queue, tempfile
os.environ.setdefault("MECABRC", r"C:\mecabkodic\mecabrc")  # 안전장치(없어도 동작)
import numpy as np
import soundfile as sf
import sounddevice as sd
from melo.api import TTS
from dolswe.audio_utils import amplitude_envelope

class TtsWorker:
    def __init__(self, on_amplitude=None, on_done=None):
        self._on_amplitude = on_amplitude or (lambda e: None)
        self._on_done = on_done or (lambda: None)
        self._model = TTS(language="KR", device="cpu")
        self._spk = self._model.hps.data.spk2id["KR"]
        self._tmp = os.path.join(tempfile.gettempdir(), "dolswe_tts.wav")
        self._q = queue.Queue()
        self._running = False
        self._interrupt = threading.Event()
        self._thread = None
        self._synth("안녕")  # 워밍업 (첫 호출 ~8s 지연 흡수)

    def _synth(self, text):
        self._model.tts_to_file(text, self._spk, self._tmp, speed=1.0, quiet=True)
        samples, rate = sf.read(self._tmp, dtype="int16")
        return samples, rate

    def start(self):
        self._running = True
        self._thread = threading.Thread(target=self._loop, daemon=True)
        self._thread.start()

    def stop(self):
        self._running = False

    def say(self, text):
        self._q.put(text)

    def stop_current(self):
        self._interrupt.set()
        sd.stop()
        try:
            while True:
                self._q.get_nowait()
        except queue.Empty:
            pass

    def _loop(self):
        while self._running:
            try:
                text = self._q.get(timeout=0.3)
            except queue.Empty:
                continue
            self._interrupt.clear()
            try:
                samples, rate = self._synth(text)
            except Exception as e:
                print("TTS 합성 실패:", e)
                continue
            if self._interrupt.is_set():
                continue
            self._on_amplitude(amplitude_envelope(samples, n_bins=20))
            sd.play(samples, rate)
            sd.wait()
            if self._q.empty():
                self._on_done()

```python
# dolswe/app.py
import threading, queue
import webview
import cv2
from dolswe import config
from dolswe.shared_state import SharedContext
from dolswe.brain import Brain
from dolswe.tts import TtsWorker
from dolswe.stt import SttWorker
from dolswe.vision_fast import VisionFast
from dolswe.vision_slow import VisionSlow

class Dolswe:
    def __init__(self):
        self.ctx = SharedContext()
        self.brain = Brain()
        self.input_q = queue.Queue(maxsize=config.INPUT_QUEUE_MAX)
        self.window = None
        self.cap = cv2.VideoCapture(0)
        self.tts = TtsWorker(on_amplitude=self._on_amp,
                             on_done=lambda: self._js("dolswe.setSpeaking(false)"))
        self.vision_slow = VisionSlow(self.ctx, self._frame)
        self.vision_fast = None
        self.stt = None

    def _frame(self):
        if not self.cap or not self.cap.isOpened():
            return None
        ok, f = self.cap.read()
        return f if ok else None

    def _js(self, code):
        if self.window:
            self.window.evaluate_js(code)

    def _on_amp(self, env):
        level = max(env) if env else 0.0
        self._js(f"dolswe.setSpeaking(true); dolswe.setAmplitude({level:.3f})")

    # --- 입력 경로 ---
    def on_user_text(self, text):          # pywebview API (타이핑)
        self._enqueue(text)

    def _on_stt_final(self, text):
        self.tts.stop_current()            # 바지인
        self._js(f"dolswe.setUserCaption({text!r})")
        self._enqueue(text)

    def _on_stt_partial(self, text):
        self._js(f"dolswe.setUserCaption({text!r})")

    def _on_new_person(self):
        self.vision_slow.describe_async()
        self._enqueue("(새 방문객이 다가옴. 먼저 말을 걸어라.)")

    def _enqueue(self, text):
        try:
            self.input_q.put_nowait(text)
        except queue.Full:
            pass

    # --- 응답 루프 ---
    def _respond_loop(self):
        while True:
            text = self.input_q.get()
            snap = self.ctx.snapshot()
            for sentence in self.brain.respond(text, snap):
                self._js(f"dolswe.setBotCaption({sentence!r})")
                self.tts.say(sentence)

    def start_workers(self):
        self.tts.start()
        if self.cap.isOpened():
            self.vision_fast = VisionFast(self.ctx, self._frame, self._on_new_person)
            self.vision_fast.start()
        try:
            self.stt = SttWorker(self._on_stt_partial, self._on_stt_final)
            self.stt.start()
        except Exception as e:
            print("STT 비활성(마이크 없음?):", e)
        threading.Thread(target=self._respond_loop, daemon=True).start()

class _Api:
    def __init__(self, app):
        self._app = app
    def on_user_text(self, text):
        self._app.on_user_text(text)

def main():
    app = Dolswe()
    api = _Api(app)
    window = webview.create_window("돌쇠", "dolswe/ui/index.html",
                                   js_api=api, fullscreen=True)
    app.window = window
    webview.start(app.start_workers)

if __name__ == "__main__":
    main()
```
